    <script>
        document.querySelectorAll("#portfolioTabs li").forEach(tab => {
            tab.addEventListener("click", function () {
                // Remove active class from all tabs
                document.querySelectorAll("#portfolioTabs li").forEach(t => t.classList.remove("active"));

                // Hide all content sections
                document.querySelectorAll(".tab-content").forEach(content => content.classList.remove("active-tab"));

                // Activate clicked tab and corresponding content
                this.classList.add("active");
                const tabId = this.getAttribute("data-tab");
                document.getElementById(`tab-${tabId}`).classList.add("active-tab");
            });
        });
    </script>
